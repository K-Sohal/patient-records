```mysql
USE PatientRecords


CREATE TABLE GP 
(
GPID int IDENTITY PRIMARY KEY NOT NULL,
FirstName VARCHAR (30) NOT NULL,
LastName VARCHAR(30) NOT NULL,
GPEmail VARCHAR(50) NOT NULL,
GPPassword VARCHAR(30) NOT NULL
);

CREATE TABLE Patient 
(
PatientID int IDENTITY PRIMARY KEY NOT NULL,
GPID int FOREIGN KEY REFERENCES GP(GPID) NOT NULL,
FirstName VARCHAR (30) NOT NULL,
LastName VARCHAR(30) NOT NULL,
DateOfBirth DATETIME NOT NULL,
Address1 VARCHAR(120) NOT NULL,
Address2 VARCHAR(120),
Address3 VARCHAR(120),
City VARCHAR(100) NOT NULL,
Region VARCHAR(100) NOT NULL,
PostalCode VARCHAR(8) NOT NULL,
ContactNumber VARCHAR(15) NOT NULL
);

CREATE TABLE Concerns
(
ConcernID int IDENTITY PRIMARY KEY NOT NULL,
PatientID int FOREIGN KEY REFERENCES Patient(PatientID) NOT NULL,
Concern VARCHAR(MAX) NOT NULL,
ConcernDate DATETIME
);

CREATE TABLE Vaccine
(
VaccineID int IDENTITY PRIMARY KEY NOT NULL,
PatientID int FOREIGN KEY REFERENCES Patient(PatientID) NOT NULL,
Vaccine VARCHAR(30) NOT NULL,
VaccineDate DATETIME NOT NULL
);


CREATE TABLE Allergy
(
AllergyID int IDENTITY PRIMARY KEY NOT NULL,
PatientID int FOREIGN KEY REFERENCES Patient(PatientID) NOT NULL,
Allergen VARCHAR(30) NOT NULL,
ReactionType VARCHAR(30) NOT NULL
);


CREATE TABLE Medication
(
MedicationID int IDENTITY PRIMARY KEY NOT NULL,
PatientID int FOREIGN KEY REFERENCES Patient(PatientID) NOT NULL,
MedicationName VARCHAR(255) NOT NULL,
Frequency VARCHAR(255) NOT NULL,
Dosage VARCHAR(255) NOT NULL,
StartDate DATETIME NOT NULL,
StopDate DATETIME NOT NULL
);
```
